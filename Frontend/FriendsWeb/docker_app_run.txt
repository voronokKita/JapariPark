# Networking
docker network create --driver=bridge friends_web_bridge


# Build
DOCKER_BUILDKIT=1 \
    docker build \
        --secret id=friends_web_user_pass,src=./secrets/.docker-build.secret \
        --file Dockerfile \
        --tag friends_web_app \
        .


# Run
docker run --detach \
        -it \
        --name FriendsWeb \
        --restart=on-failure:2 \
        --mount type=bind,src="$(pwd)"/logs,dst=/FriendsWeb/logs \
        --network friends_web_bridge \
        --network-alias friends-web.app \
        friends_web_app
